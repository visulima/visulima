# Examples

This page provides practical examples for common use cases with `@visulima/tabular`.

## Basic Examples

### Simple Data Table

Display tabular data with headers and rows:

```typescript title="simple-data-table.ts"
import { createTable } from "@visulima/tabular";

const users = [
  { name: "Alice Johnson", email: "alice@example.com", role: "Admin" },
  { name: "Bob Smith", email: "bob@example.com", role: "User" },
  { name: "Charlie Brown", email: "charlie@example.com", role: "Editor" }
];

const table = createTable();
table.setHeaders(["Name", "Email", "Role"]);

users.forEach(user => {
  table.addRow([user.name, user.email, user.role]);
});

console.log(table.toString());
```

```typescript
console.log(table.toString());
```

## CLI Tool Examples

### Server Status Dashboard

Monitor server health:

```typescript
import { createTable } from "@visulima/tabular";

const table = createTable({
  style: {
    border: ROUNDED_BORDER,
    paddingLeft: 2,
    paddingRight: 2
  }
});

table.setHeaders([
  { content: bold("Server"), hAlign: "left" },
  { content: bold("Status"), hAlign: "center" },
  { content: bold("CPU"), hAlign: "center" },
  { content: bold("Memory"), hAlign: "center" },
  { content: bold("Uptime"), hAlign: "right" }
]);

table.addRows(
  [
    "api-server-01",
    { backgroundColor: bgGreen, content: white("● Online") },
    green("28%"),
    green("45%"),
    "24d 12h"
  ],
  [
    "api-server-02",
    { backgroundColor: bgYellow, content: white("⚠ Warning") },
    yellow("78%"),
    yellow("82%"),
    "15d 6h"
  ],
  [
    "cache-server",
    { backgroundColor: bgRed, content: white("✖ Down") },
    red("95%"),
```

```typescript
);

console.log(table.toString());
```

## Data Visualization

### Progress Tracking

Show progress bars in tables:

```typescript
import { createTable } from "@visulima/tabular";
import { green, yellow, red } from "@visulima/colorize";

function progressBar(percent: number): string {
  const width = 20;
  const filled = Math.round(width * (percent / 100));
  const bar = "█".repeat(filled) + "░".repeat(width - filled);
  
  let color = green;
  if (percent < 50) color = red;
  else if (percent < 80) color = yellow;
  
  return color(bar);
}

const table = createTable();

table.setHeaders(["Task", "Progress", "Status"]);

table.addRows(
  ["Data Migration", progressBar(100), green("Complete")],
  ["API Integration", progressBar(65), yellow("In Progress")],
  ["Testing", progressBar(30), red("Blocked")]
);

console.log(table.toString());
```

### Dashboard Layout with Grid

Create a complex dashboard:

```typescript
import { createGrid } from "@visulima/tabular";
import { ROUNDED_BORDER } from "@visulima/tabular/style";
import { blue, bold, green, yellow } from "@visulima/colorize";

const grid = createGrid({
  columns: 2,
  gap: 1,
  border: ROUNDED_BORDER,
  paddingLeft: 2,
  paddingRight: 2
});

grid.addItems([
  { colSpan: 2, content: bold(blue("System Dashboard")), hAlign: "center" },
  { content: `${bold("CPU Usage")}\n${green("45%")}`, vAlign: "middle" },
  { content: `${bold("Memory")}\n${yellow("78%")}`, vAlign: "middle" },
  { content: `${bold("Disk I/O")}\n${green("12 MB/s")}`, vAlign: "middle" },
  { content: `${bold("Network")}\n${green("350 KB/s")}`, vAlign: "middle" }
]);

console.log(grid.toString());
```

## Responsive Tables

### Word Wrapping for Long Content

Handle long text gracefully:

```typescript
import { createTable } from "@visulima/tabular";

const table = createTable({
  maxWidth: 80,
  wordWrap: true
});

table.setHeaders(["Command", "Description"]);

table.addRows(
  [
    "install",
    "Installs all dependencies listed in package.json. Creates node_modules directory if it doesn't exist."
  ],
  [
    "update",
    "Updates all dependencies to their latest versions according to the version ranges specified in package.json."
  ],
  [
    "build",
    "Compiles TypeScript code to JavaScript and bundles all assets for production deployment."
  ]
);

console.log(table.toString());
```

### Mixed Wrapping and Truncation

Combine word wrapping with truncation:

```typescript
import { createTable } from "@visulima/tabular";

const table = createTable();

table.setHeaders(["Title", "Description", "Status"]);

table.addRow([
  { content: "Feature Implementation", maxWidth: 15, truncate: true },

    content: "Implement the new authentication system with OAuth2 support and multi-factor authentication",
    wordWrap: true,
    maxWidth: 40
  },
  "In Progress"

```

## Special Use Cases

### Configuration Display

Show configuration values:

```typescript
import { createTable } from "@visulima/tabular";
import { NO_BORDER } from "@visulima/tabular/style";
import { blue, bold } from "@visulima/colorize";

const table = createTable({
  style: {
    border: NO_BORDER,
    paddingLeft: 0,
    paddingRight: 2
  }
});

const config = {
  "Server Port": "3000",
  "Database URL": "postgresql://localhost:5432/mydb",
  "API Key": "*********************",
  "Log Level": "debug",
  "Max Connections": "100"
};

Object.entries(config).forEach(([key, value]) => {
  table.addRow([bold(blue(key)), value]);
});

console.log(table.toString());
```

## Integration Examples

### Using with Inquirer

Combine with CLI prompts:

```typescript
import { createTable } from "@visulima/tabular";
import inquirer from "inquirer";

const users = [
  { id: 1, name: "Alice", role: "Admin" },
  { id: 2, name: "Bob", role: "User" },
  { id: 3, name: "Charlie", role: "Editor" }
];

// Display users
const table = createTable();
table.setHeaders(["ID", "Name", "Role"]);
users.forEach(user => {
  table.addRow([user.id.toString(), user.name, user.role]);
});
console.log(table.toString());

// Prompt for selection
const { userId } = await inquirer.prompt([
  {
    type: "input",
    name: "userId",
    message: "Select user ID:"
  }
]);
```

### Live Updates

Update table content in real-time:

```typescript
import { createTable } from "@visulima/tabular";

function clearScreen() {
  process.stdout.write("\u001B[2J\u001B[0;0f");
}

function displayMetrics(cpu: number, memory: number, requests: number) {
  clearScreen();
  
  const table = createTable();
  table.setHeaders(["Metric", "Value"]);
  table.addRows(
    ["CPU Usage", `${cpu}%`],
    ["Memory Usage", `${memory}%`],
    ["Requests/sec", requests.toString()]
  );
  
  console.log(table.toString());
}

// Simulate live updates
let cpu = 45, memory = 60, requests = 120;
setInterval(() => {
  cpu = Math.max(0, Math.min(100, cpu + Math.random() * 10 - 5));
  memory = Math.max(0, Math.min(100, memory + Math.random() * 5 - 2.5));
  requests = Math.max(0, requests + Math.floor(Math.random() * 20 - 10));
  
  displayMetrics(
    Math.round(cpu),
    Math.round(memory),
    requests
  );
}, 1000);
```
